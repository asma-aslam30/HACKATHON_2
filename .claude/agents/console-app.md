### 1. In-Memory Task Storage

**Primary Responsibility**: Implement simple in-memory data structure for storing todos during CLI session.

**Data Structure**:

```python
# backend/src/cli/storage.py

from typing import List, Dict, Any, Optional
from datetime import datetime

class InMemoryStorage:
    """Simple in-memory storage for todos (Phase I only)"""

    def __init__(self):
        self.todos: List[Dict[str, Any]] = []
        self.next_id: int = 1

    def add_todo(
        self,
        title: str,
        priority: str = "medium",
        tags: Optional[List[str]] = None,
        due_date: Optional[datetime] = None
    ) -> Dict[str, Any]:
        """Add a new todo and return it"""
        todo = {
            "id": self.next_id,
            "title": title,
            "completed": False,
            "priority": priority,
            "tags": tags or [],
            "due_date": due_date,
            "created_at": datetime.now(),
            "updated_at": datetime.now()
        }
        self.todos.append(todo)
        self.next_id += 1
        return todo

    def get_todos(
        self,
        completed: Optional[bool] = None,
        priority: Optional[str] = None,
        tags: Optional[List[str]] = None
    ) -> List[Dict[str, Any]]:
        """Get todos with optional filters"""
        filtered_todos = self.todos

        if completed is not None:
            filtered_todos = [t for t in filtered_todos if t["completed"] == completed]

        if priority:
            filtered_todos = [t for t in filtered_todos if t["priority"] == priority]

        if tags:
            filtered_todos = [
                t for t in filtered_todos
                if any(tag in t["tags"] for tag in tags)
            ]

        return filtered_todos

    def get_todo_by_id(self, todo_id: int) -> Optional[Dict[str, Any]]:
        """Get a single todo by ID"""
        for todo in self.todos:
            if todo["id"] == todo_id:
                return todo
        return None

    def update_todo(self, todo_id: int, **updates) -> Optional[Dict[str, Any]]:
        """Update a todo and return it"""
        todo = self.get_todo_by_id(todo_id)
        if not todo:
            return None

        for key, value in updates.items():
            if key in todo:
                todo[key] = value

        todo["updated_at"] = datetime.now()
        return todo

    def delete_todo(self, todo_id: int) -> bool:
        """Delete a todo, return True if found and deleted"""
        todo = self.get_todo_by_id(todo_id)
        if todo:
            self.todos.remove(todo)
            return True
        return False

    def complete_todo(self, todo_id: int) -> Optional[Dict[str, Any]]:
        """Mark a todo as complete"""
        return self.update_todo(todo_id, completed=True)

    def search_todos(self, query: str) -> List[Dict[str, Any]]:
        """Search todos by title"""
        query_lower = query.lower()
        return [
            t for t in self.todos
            if query_lower in t["title"].lower()
        ]
```

**Key Features**:
- Simple list-based storage (todos lost when CLI exits)
- Auto-incrementing IDs
- Support for filtering (completed, priority, tags)
- Support for search by title
- Timestamps (created_at, updated_at)

```python
#!/usr/bin/env python3
"""
Evolution of Todo - Phase I CLI Application

Commands:
  add       Add a new todo
  list      List all todos
  view      View details of a todo
  complete  Mark a todo as complete
  delete    Delete a todo
  search    Search todos by title
  filter    Filter todos by criteria
"""

import argparse
import sys
from datetime import datetime
from typing import Optional, List
from .storage import InMemoryStorage
from .display import TodoDisplay
from .parser import parse_due_date, parse_tags

# Global storage instance (in-memory for Phase I)
storage = InMemoryStorage()
display = TodoDisplay()


def cmd_add(args):
    """Add a new todo"""
    try:
        # Parse due date if provided
        due_date = parse_due_date(args.due) if args.due else None

        # Parse tags
        tags = parse_tags(args.tags) if args.tags else []

        # Add todo
        todo = storage.add_todo(
            title=args.title,
            priority=args.priority,
            tags=tags,
            due_date=due_date
        )

        display.success(f"Todo added: {todo['title']} (ID: {todo['id']})")
        display.show_todo(todo)

    except ValueError as e:
        display.error(f"Invalid input: {e}")
        sys.exit(1)


def cmd_list(args):
    """List all todos"""
    # Apply filters
    completed = None
    if args.completed:
        completed = True
    elif args.active:
        completed = False

    todos = storage.get_todos(
        completed=completed,
        priority=args.priority
    )

    if not todos:
        display.info("No todos found.")
        return

    display.show_todo_list(todos, title=f"Todos ({len(todos)})")


def cmd_view(args):
    """View details of a todo"""
    todo = storage.get_todo_by_id(args.id)

    if not todo:
        display.error(f"Todo with ID {args.id} not found")
        sys.exit(1)

    display.show_todo_detail(todo)


def cmd_complete(args):
    """Mark a todo as complete"""
    todo = storage.complete_todo(args.id)

    if not todo:
        display.error(f"Todo with ID {args.id} not found")
        sys.exit(1)

    display.success(f"Todo marked as complete: {todo['title']}")
    display.show_todo(todo)


def cmd_delete(args):
    """Delete a todo"""
    todo = storage.get_todo_by_id(args.id)

    if not todo:
        display.error(f"Todo with ID {args.id} not found")
        sys.exit(1)

    # Confirm deletion
    if not args.force:
        confirm = input(f"Delete todo '{todo['title']}'? (y/N): ")
        if confirm.lower() != 'y':
            display.info("Deletion cancelled")
            return

    storage.delete_todo(args.id)
    display.success(f"Todo deleted: {todo['title']}")


def cmd_search(args):
    """Search todos by title"""
    todos = storage.search_todos(args.query)

    if not todos:
        display.info(f"No todos found matching '{args.query}'")
        return

    display.show_todo_list(todos, title=f"Search Results ({len(todos)})")


def cmd_filter(args):
    """Filter todos by criteria"""
    # Parse tags
    tags = parse_tags(args.tags) if args.tags else None

    todos = storage.get_todos(
        completed=args.completed,
        priority=args.priority,
        tags=tags
    )

    if not todos:
        display.info("No todos match the filter criteria")
        return

    display.show_todo_list(todos, title=f"Filtered Todos ({len(todos)})")


def create_parser():
    """Create argument parser"""
    parser = argparse.ArgumentParser(
        description="Evolution of Todo - Phase I CLI",
        formatter_class=argparse.RawDescriptionHelpFormatter
    )

    subparsers = parser.add_subparsers(dest="command", help="Available commands")

    # Add command
    parser_add = subparsers.add_parser("add", help="Add a new todo")
    parser_add.add_argument("title", help="Todo title")
    parser_add.add_argument("-p", "--priority", choices=["low", "medium", "high"],
                           default="medium", help="Priority (default: medium)")
    parser_add.add_argument("-t", "--tags", help="Comma-separated tags")
    parser_add.add_argument("-d", "--due", help="Due date (e.g., '2025-12-25', 'tomorrow', 'next week')")
    parser_add.set_defaults(func=cmd_add)

    # List command
    parser_list = subparsers.add_parser("list", help="List all todos")
    parser_list.add_argument("-c", "--completed", action="store_true",
                            help="Show only completed todos")
    parser_list.add_argument("-a", "--active", action="store_true",
                            help="Show only active todos")
    parser_list.add_argument("-p", "--priority", choices=["low", "medium", "high"],
                            help="Filter by priority")
    parser_list.set_defaults(func=cmd_list)

    # View command
    parser_view = subparsers.add_parser("view", help="View todo details")
    parser_view.add_argument("id", type=int, help="Todo ID")
    parser_view.set_defaults(func=cmd_view)

    # Complete command
    parser_complete = subparsers.add_parser("complete", help="Mark todo as complete")
    parser_complete.add_argument("id", type=int, help="Todo ID")
    parser_complete.set_defaults(func=cmd_complete)

    # Delete command
    parser_delete = subparsers.add_parser("delete", help="Delete a todo")
    parser_delete.add_argument("id", type=int, help="Todo ID")
    parser_delete.add_argument("-f", "--force", action="store_true",
                              help="Skip confirmation")
    parser_delete.set_defaults(func=cmd_delete)

    # Search command
    parser_search = subparsers.add_parser("search", help="Search todos by title")
    parser_search.add_argument("query", help="Search query")
    parser_search.set_defaults(func=cmd_search)

    # Filter command
    parser_filter = subparsers.add_parser("filter", help="Filter todos by criteria")
    parser_filter.add_argument("-c", "--completed", action="store_true",
                              help="Show only completed todos")
    parser_filter.add_argument("-p", "--priority", choices=["low", "medium", "high"],
                              help="Filter by priority")
    parser_filter.add_argument("-t", "--tags", help="Filter by tags (comma-separated)")
    parser_filter.set_defaults(func=cmd_filter)

    return parser


def main():
    """Main entry point"""
    parser = create_parser()
    args = parser.parse_args()

    if not args.command:
        parser.print_help()
        sys.exit(1)

    # Execute command
    args.func(args)


if __name__ == "__main__":
    main()
```

from datetime import datetime, timedelta
from typing import List, Optional
import re

def parse_tags(tags_str: str) -> List[str]:
    """Parse comma-separated tags"""
    if not tags_str:
        return []

    return [tag.strip() for tag in tags_str.split(",") if tag.strip()]


def parse_due_date(date_str: str) -> Optional[datetime]:
    """
    Parse due date from natural language or ISO format

    Supports:
    - ISO format: 2025-12-25
    - Natural language: today, tomorrow, next week, in 3 days
    """
    date_str = date_str.lower().strip()

    # Today
    if date_str == "today":
        return datetime.now().replace(hour=23, minute=59, second=59)

    # Tomorrow
    if date_str == "tomorrow":
        return (datetime.now() + timedelta(days=1)).replace(hour=23, minute=59, second=59)

    # Next week
    if date_str == "next week":
        return (datetime.now() + timedelta(weeks=1)).replace(hour=23, minute=59, second=59)

    # In X days/weeks/months
    match = re.match(r"in (\d+) (day|week|month)s?", date_str)
    if match:
        amount = int(match.group(1))
        unit = match.group(2)

        if unit == "day":
            return (datetime.now() + timedelta(days=amount)).replace(hour=23, minute=59, second=59)
        elif unit == "week":
            return (datetime.now() + timedelta(weeks=amount)).replace(hour=23, minute=59, second=59)
        elif unit == "month":
            return (datetime.now() + timedelta(days=amount * 30)).replace(hour=23, minute=59, second=59)

    # ISO format: YYYY-MM-DD or YYYY-MM-DD HH:MM
    try:
        return datetime.fromisoformat(date_str)
    except ValueError:
        pass

    raise ValueError(f"Cannot parse due date: {date_str}")


def validate_priority(priority: str) -> str:
    """Validate priority level"""
    priority = priority.lower()
    if priority not in ["low", "medium", "high"]:
        raise ValueError(f"Priority must be low, medium, or high (got: {priority})")
    return priority
```

# backend/src/cli/reminders.py

from datetime import datetime, timedelta
from typing import Dict, Any
import threading
import time

class ReminderService:
    """Simple reminder service for due todos"""

    def __init__(self, storage):
        self.storage = storage
        self.running = False
        self.thread = None

    def start(self):
        """Start reminder background thread"""
        self.running = True
        self.thread = threading.Thread(target=self._check_reminders, daemon=True)
        self.thread.start()

    def stop(self):
        """Stop reminder service"""
        self.running = False
        if self.thread:
            self.thread.join()

    def _check_reminders(self):
        """Background thread to check for due todos"""
        while self.running:
            now = datetime.now()

            # Check todos due in next hour
            todos = self.storage.get_todos(completed=False)
            for todo in todos:
                if todo.get("due_date"):
                    time_until_due = todo["due_date"] - now

                    # Remind if due in next hour and not reminded yet
                    if timedelta(0) < time_until_due < timedelta(hours=1):
                        if not todo.get("reminded"):
                            self._send_notification(todo)
                            self.storage.update_todo(todo["id"], reminded=True)

            # Check every 5 minutes
            time.sleep(300)

    def _send_notification(self, todo: Dict[str, Any]):
        """Send console notification"""
        from .display import TodoDisplay
        display = TodoDisplay()

        print("\n" + "=" * 50)
        display.warning(f"⏰ REMINDER: '{todo['title']}' is due soon!")
        print("=" * 50 + "\n")

"""
Voice input for CLI (bonus feature)

Requires: pip install SpeechRecognition pyaudio
"""

import sys

try:
    import speech_recognition as sr
except ImportError:
    print("Voice input requires: pip install SpeechRecognition pyaudio")
    sys.exit(1)


class VoiceInput:
    """Voice-to-text input for todos"""

    def __init__(self):
        self.recognizer = sr.Recognizer()

    def listen(self, prompt: str = "Speak your todo...") -> str:
        """
        Listen to microphone and return transcribed text

        Returns:
            Transcribed text or empty string if failed
        """
        print(prompt)
        print("🎤 Listening... (speak now)")

        with sr.Microphone() as source:
            # Adjust for ambient noise
            self.recognizer.adjust_for_ambient_noise(source, duration=0.5)

            try:
                # Listen with timeout
                audio = self.recognizer.listen(source, timeout=5, phrase_time_limit=10)
                print("Processing...")

                # Recognize speech using Google Speech Recognition
                text = self.recognizer.recognize_google(audio)
                print(f"✓ Heard: {text}")
                return text

            except sr.WaitTimeoutError:
                print("✗ No speech detected (timeout)")
                return ""

            except sr.UnknownValueError:
                print("✗ Could not understand audio")
                return ""

            except sr.RequestError as e:
                print(f"✗ Speech recognition error: {e}")
                return ""


def add_voice_command_to_parser(subparsers):
    """Add 'voice' subcommand to argparse parser"""
    parser_voice = subparsers.add_parser("voice", help="Add todo via voice input")
    parser_voice.add_argument("-p", "--priority", choices=["low", "medium", "high"],
                             default="medium", help="Priority (default: medium)")
    parser_voice.set_defaults(func=cmd_voice)


def cmd_voice(args):
    """Voice input command handler"""
    from .storage import storage
    from .display import display

    voice = VoiceInput()
    title = voice.listen("What do you need to do?")

    if not title:
        display.error("Failed to capture voice input")
        sys.exit(1)

    # Add todo with voice input
    todo = storage.add_todo(
        title=title,
        priority=args.priority
    )

    display.success(f"Todo added from voice: {todo['title']} (ID: {todo['id']})")
    display.show_todo(todo)

"""
Console display utilities with colors and formatting
"""

from typing import List, Dict, Any
from datetime import datetime
from rich.console import Console
from rich.table import Table
from rich.panel import Panel
from rich import box

console = Console()

# ANSI color codes
class Colors:
    RESET = "\033[0m"
    RED = "\033[91m"
    GREEN = "\033[92m"
    YELLOW = "\033[93m"
    BLUE = "\033[94m"
    MAGENTA = "\033[95m"
    CYAN = "\033[96m"
    WHITE = "\033[97m"
    BOLD = "\033[1m"
    DIM = "\033[2m"


class TodoDisplay:
    """Display utilities for todos"""

    def __init__(self):
        self.use_rich = True  # Use rich library for better formatting

    def success(self, message: str):
        """Print success message"""
        console.print(f"✓ {message}", style="bold green")

    def error(self, message: str):
        """Print error message"""
        console.print(f"✗ {message}", style="bold red")

    def warning(self, message: str):
        """Print warning message"""
        console.print(f"⚠ {message}", style="bold yellow")

    def info(self, message: str):
        """Print info message"""
        console.print(f"ℹ {message}", style="cyan")

    def show_todo(self, todo: Dict[str, Any]):
        """Display a single todo (compact format)"""
        status = "✓" if todo["completed"] else "☐"
        priority_icon = self._get_priority_icon(todo["priority"])

        title = todo["title"]
        if todo["completed"]:
            title = f"[dim strikethrough]{title}[/dim strikethrough]"

        tags_str = ""
        if todo["tags"]:
            tags_str = " " + " ".join([f"[cyan]#{tag}[/cyan]" for tag in todo["tags"]])

        due_str = ""
        if todo.get("due_date"):
            due_str = f" [yellow]Due: {self._format_due_date(todo['due_date'])}[/yellow]"

        console.print(f"{status} [{todo['id']}] {priority_icon} {title}{tags_str}{due_str}")

    def show_todo_detail(self, todo: Dict[str, Any]):
        """Display detailed todo information"""
        status = "✓ Completed" if todo["completed"] else "☐ Active"
        priority = todo["priority"].title()

        # Build details dict
        details = {
            "ID": str(todo["id"]),
            "Status": status,
            "Title": todo["title"],
            "Priority": f"{self._get_priority_icon(todo['priority'])} {priority}",
            "Tags": ", ".join(todo["tags"]) if todo["tags"] else "None",
            "Created": self._format_datetime(todo["created_at"]),
            "Updated": self._format_datetime(todo["updated_at"])
        }

        if todo.get("due_date"):
            details["Due Date"] = self._format_datetime(todo["due_date"])

        # Create panel
        panel_content = "\n".join([f"[bold]{k}:[/bold] {v}" for k, v in details.items()])
        panel = Panel(panel_content, title="Todo Details", border_style="blue", box=box.ROUNDED)
        console.print(panel)

    def show_todo_list(self, todos: List[Dict[str, Any]], title: str = "Todos"):
        """Display list of todos in table format"""
        if not todos:
            self.info("No todos to display")
            return

        table = Table(title=title, box=box.ROUNDED)
        table.add_column("ID", style="cyan", width=4)
        table.add_column("Status", width=6)
        table.add_column("Priority", width=8)
        table.add_column("Title", style="white")
        table.add_column("Tags", style="dim")
        table.add_column("Due", style="yellow")

        for todo in todos:
            status = "✓" if todo["completed"] else "☐"
            priority = f"{self._get_priority_icon(todo['priority'])} {todo['priority']}"
            title_text = todo["title"]
            if todo["completed"]:
                title_text = f"[dim strikethrough]{title_text}[/dim strikethrough]"

            tags = ", ".join([f"#{t}" for t in todo["tags"]]) if todo["tags"] else ""
            due = self._format_due_date(todo["due_date"]) if todo.get("due_date") else ""

            table.add_row(
                str(todo["id"]),
                status,
                priority,
                title_text,
                tags,
                due
            )

        console.print(table)

    def _get_priority_icon(self, priority: str) -> str:
        """Get icon for priority"""
        icons = {
            "low": "🟢",
            "medium": "🟡",
            "high": "🔴"
        }
        return icons.get(priority, "⚪")

    def _format_datetime(self, dt: datetime) -> str:
        """Format datetime for display"""
        return dt.strftime("%Y-%m-%d %H:%M")

    def _format_due_date(self, dt: datetime) -> str:
        """Format due date with relative time"""
        now = datetime.now()
        delta = dt - now

        if delta.days < 0:
            return f"Overdue ({abs(delta.days)}d)"
        elif delta.days == 0:
            return "Today"
        elif delta.days == 1:
            return "Tomorrow"
        elif delta.days < 7:
            return f"In {delta.days}d"
        else:
            return dt.strftime("%b %d")
```
