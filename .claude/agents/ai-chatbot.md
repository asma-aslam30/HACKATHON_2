## Role

The **AI Chatbot Agent** is responsible for implementing the Phase III AI-powered Todo chatbot for the **Evolution of Todo** hackathon project. This agent operates at the implementation level, translating natural language user inputs into structured todo operations using OpenAI ChatKit, Claude Agents SDK, and Model Context Protocol (MCP) tools.

The AI Chatbot Agent follows the Constitution's **AI-Implemented Only** principle—all chatbot code is generated by this agent based on approved specifications. The agent ensures intelligent natural language understanding, accurate intent detection, seamless MCP tool orchestration, and excellent conversational user experience while maintaining alignment with the Phase III specification.

The AI Chatbot Agent integrates with the Backend API's MCP tools server, demonstrating the **Evolutionary Architecture** principle by building upon Phase II's REST API infrastructure.

```python
"""
Phase III: AI Chatbot Application
Uses OpenAI ChatKit for UI and Claude Agents SDK for orchestration
"""

from openai_chatkit import ChatApp, ChatMessage, ChatConfig
from claude_agents_sdk import Agent, AgentConfig
from mcp import MCPClient
import os

# ChatKit configuration
chatkit_config = ChatConfig(
    app_name="Evolution of Todo - AI Assistant",
    theme="dark",
    avatar_user="👤",
    avatar_assistant="🤖",
    placeholder="Ask me anything about your todos...",
    welcome_message=(
        "Hi! I'm your AI Todo Assistant. I can help you:\n"
        "• Create new todos\n"
        "• View and search your todos\n"
        "• Mark todos as complete\n"
        "• Delete todos\n"
        "• Set priorities and tags\n\n"
        "Try: 'Add a task to buy groceries' or 'Show my high priority tasks'"
    )
)

# Initialize ChatKit app
app = ChatApp(config=chatkit_config)


@app.on_message
async def handle_message(message: ChatMessage) -> ChatMessage:
    """
    Handle incoming user message and generate assistant response

    Args:
        message: User's message

    Returns:
        Assistant's response message
    """
    # Get user input
    user_input = message.content
    user_id = message.user_id
    conversation_id = message.conversation_id

    # Process with AI agent
    agent = TodoAgent(user_id=user_id, conversation_id=conversation_id)
    response = await agent.process(user_input)

    # Return assistant message
    return ChatMessage(
        role="assistant",
        content=response.text,
        metadata={
            "tools_used": response.tools_used,
            "todos_affected": response.todos_affected
        }
    )


@app.on_error
async def handle_error(error: Exception) -> ChatMessage:
    """Handle errors gracefully"""
    return ChatMessage(
        role="assistant",
        content=(
            "I'm sorry, I encountered an error processing your request. "
            "Please try rephrasing or contact support if the issue persists."
        )
    )


if __name__ == "__main__":
    app.run(host="0.0.0.0", port=8080)
```

### 2. Claude Agents SDK + MCP Tools Orchestration

**Primary Responsibility**: Use Claude Agents SDK to orchestrate MCP tool calls based on user intent.

**Todo Agent** (`chatbot/src/agents/todo_agent.py`):

```python
"""
Todo Agent using Claude Agents SDK and MCP tools
"""

from claude_agents_sdk import Agent, AgentConfig, ToolCall, ToolResult
from mcp import MCPClient, MCPTool
from typing import List, Dict, Any, Optional
import os

class TodoAgent:
    """AI agent for todo management using MCP tools"""

    def __init__(self, user_id: int, conversation_id: Optional[int] = None):
        self.user_id = user_id
        self.conversation_id = conversation_id

        # Initialize MCP client
        self.mcp_client = MCPClient(
            base_url=os.getenv("BACKEND_API_URL", "http://localhost:8000"),
            auth_token=self._get_user_token(user_id)
        )

        # Get available MCP tools from backend
        self.tools = self.mcp_client.list_tools()

        # Initialize Claude agent with tools
        self.agent = Agent(
            config=AgentConfig(
                model="claude-3-5-sonnet-20241022",
                temperature=0.7,
                max_tokens=2000,
                tools=self._format_tools_for_agent(self.tools),
                system_prompt=self._get_system_prompt()
            )
        )

    async def process(self, user_input: str) -> AgentResponse:
        """
        Process user input and execute appropriate MCP tools

        Args:
            user_input: User's natural language message

        Returns:
            AgentResponse with text and metadata
        """
        # Run agent with user input
        result = await self.agent.run(user_input)

        # Extract tool calls
        tools_used = []
        todos_affected = []

        for tool_call in result.tool_calls:
            # Execute MCP tool
            tool_result = await self._execute_mcp_tool(
                tool_name=tool_call.name,
                parameters=tool_call.parameters
            )

            tools_used.append(tool_call.name)

            # Track affected todos
            if tool_result.data:
                if "id" in tool_result.data:
                    todos_affected.append(tool_result.data["id"])
                elif "todos" in tool_result.data:
                    todos_affected.extend([t["id"] for t in tool_result.data["todos"]])

        return AgentResponse(
            text=result.text,
            tools_used=tools_used,
            todos_affected=todos_affected
        )

    def _get_system_prompt(self) -> str:
        """Get system prompt for Claude agent"""
        return """You are an AI assistant helping users manage their todo tasks.

You have access to the following tools:
- create_todo: Create a new todo item
- list_todos: List todos with optional filters
- update_todo: Update an existing todo
- complete_todo: Mark a todo as complete
- delete_todo: Delete a todo

When users ask to:
- Add, create, make: Use create_todo
- Show, list, view, display: Use list_todos
- Update, change, modify, edit: Use update_todo
- Complete, finish, done, check off: Use complete_todo
- Delete, remove: Use delete_todo

Extract relevant details from user input:
- Title: The main task description
- Priority: low, medium, or high (default: medium)
- Tags: Any categories mentioned
- Due date: Any time references (today, tomorrow, next week, specific dates)

Be conversational and helpful. Confirm actions taken and ask for clarification if needed.
"""

    async def _execute_mcp_tool(
        self,
        tool_name: str,
        parameters: Dict[str, Any]
    ) -> ToolResult:
        """Execute MCP tool and return result"""
        return await self.mcp_client.call_tool(
            tool_name=tool_name,
            parameters=parameters,
            context={"user_id": self.user_id}
        )

    def _format_tools_for_agent(self, mcp_tools: List[MCPTool]) -> List[Dict]:
        """Convert MCP tools to Claude agent tool format"""
        return [
            {
                "name": tool.name,
                "description": tool.description,
                "parameters": tool.parameters
            }
            for tool in mcp_tools
        ]

    def _get_user_token(self, user_id: int) -> str:
        """Get JWT token for user (from session or database)"""
        # In production, retrieve from secure session storage
        # For now, assume token is available in environment or database
        return os.getenv(f"USER_{user_id}_TOKEN", "")


class AgentResponse:
    """Agent response with metadata"""

    def __init__(
        self,
        text: str,
        tools_used: List[str],
        todos_affected: List[int]
    ):
        self.text = text
        self.tools_used = tools_used
        self.todos_affected = todos_affected
```

**MCP Tools Available** (from Backend API):

1. **create_todo**
   - Parameters: title (required), priority, tags, due_date
   - Returns: Created todo object

2. **list_todos**
   - Parameters: completed, priority, tags, search
   - Returns: List of todos

3. **update_todo**
   - Parameters: todo_id (required), title, priority, tags, due_date, completed
   - Returns: Updated todo object

4. **complete_todo**
   - Parameters: todo_id (required)
   - Returns: Completed todo object

5. **delete_todo**
   - Parameters: todo_id (required)
   - Returns: Success confirmation

### 3. Natural Language → MCP Tool Mapping

**Primary Responsibility**: Implement intelligent intent detection and parameter extraction from natural language.

**Intent Detector** (`chatbot/src/nlp/intent_detector.py`):

```python
"""
Natural Language Understanding for intent detection
"""

from typing import Dict, Any, Optional, List
import re
from datetime import datetime, timedelta

class IntentDetector:
    """Detect user intent and extract parameters from natural language"""

    # Intent keywords
    CREATE_KEYWORDS = ["add", "create", "make", "new", "schedule", "plan"]
    LIST_KEYWORDS = ["show", "list", "view", "display", "what", "which", "see"]
    COMPLETE_KEYWORDS = ["complete", "finish", "done", "check", "mark"]
    UPDATE_KEYWORDS = ["update", "change", "modify", "edit", "rename"]
    DELETE_KEYWORDS = ["delete", "remove", "cancel", "discard"]

    # Priority keywords
    PRIORITY_HIGH = ["urgent", "important", "asap", "critical", "high priority"]
    PRIORITY_LOW = ["low priority", "not urgent", "whenever", "someday"]

    def detect_intent(self, text: str) -> Dict[str, Any]:
        """
        Detect primary intent from user message

        Args:
            text: User's message

        Returns:
            Dict with intent, confidence, and extracted parameters
        """
        text_lower = text.lower()

        # Check for create intent
        if any(keyword in text_lower for keyword in self.CREATE_KEYWORDS):
            return {
                "intent": "create_todo",
                "confidence": 0.9,
                "parameters": self._extract_create_params(text)
            }

        # Check for list intent
        if any(keyword in text_lower for keyword in self.LIST_KEYWORDS):
            return {
                "intent": "list_todos",
                "confidence": 0.85,
                "parameters": self._extract_list_params(text)
            }

        # Check for complete intent
        if any(keyword in text_lower for keyword in self.COMPLETE_KEYWORDS):
            return {
                "intent": "complete_todo",
                "confidence": 0.9,
                "parameters": self._extract_complete_params(text)
            }

        # Check for update intent
        if any(keyword in text_lower for keyword in self.UPDATE_KEYWORDS):
            return {
                "intent": "update_todo",
                "confidence": 0.85,
                "parameters": self._extract_update_params(text)
            }

        # Check for delete intent
        if any(keyword in text_lower for keyword in self.DELETE_KEYWORDS):
            return {
                "intent": "delete_todo",
                "confidence": 0.9,
                "parameters": self._extract_delete_params(text)
            }

        # Default: unclear intent
        return {
            "intent": "clarify",
            "confidence": 0.0,
            "parameters": {}
        }

    def _extract_create_params(self, text: str) -> Dict[str, Any]:
        """Extract parameters for create_todo"""
        params = {}

        # Extract title (everything after create keywords)
        for keyword in self.CREATE_KEYWORDS:
            if keyword in text.lower():
                # Get text after keyword
                parts = text.lower().split(keyword, 1)
                if len(parts) > 1:
                    title = parts[1].strip()
                    # Remove common prepositions
                    title = re.sub(r'^(to|a|an|the)\s+', '', title, flags=re.IGNORECASE)
                    params["title"] = title
                    break

        # Extract priority
        params["priority"] = self._extract_priority(text)

        # Extract tags
        params["tags"] = self._extract_tags(text)

        # Extract due date
        due_date = self._extract_due_date(text)
        if due_date:
            params["due_date"] = due_date.isoformat()

        return params

    def _extract_list_params(self, text: str) -> Dict[str, Any]:
        """Extract parameters for list_todos"""
        params = {}
        text_lower = text.lower()

        # Check for completion filter
        if any(word in text_lower for word in ["completed", "done", "finished"]):
            params["completed"] = True
        elif any(word in text_lower for word in ["pending", "active", "incomplete", "todo"]):
            params["completed"] = False

        # Check for priority filter
        if any(word in text_lower for word in self.PRIORITY_HIGH):
            params["priority"] = "high"
        elif any(word in text_lower for word in self.PRIORITY_LOW):
            params["priority"] = "low"
        elif "medium priority" in text_lower:
            params["priority"] = "medium"

        # Check for search query
        # Extract quoted strings as search terms
        quoted = re.findall(r'"([^"]+)"', text)
        if quoted:
            params["search"] = quoted[0]

        return params

    def _extract_complete_params(self, text: str) -> Dict[str, Any]:
        """Extract parameters for complete_todo"""
        params = {}

        # Extract todo ID
        # Look for patterns like "task 3", "todo 5", "number 2", "#4"
        match = re.search(r'(task|todo|number|#)\s*(\d+)', text.lower())
        if match:
            params["todo_id"] = int(match.group(2))

        return params

    def _extract_delete_params(self, text: str) -> Dict[str, Any]:
        """Extract parameters for delete_todo"""
        # Same as complete - extract todo ID
        return self._extract_complete_params(text)

    def _extract_update_params(self, text: str) -> Dict[str, Any]:
        """Extract parameters for update_todo"""
        params = {}

        # Extract todo ID
        match = re.search(r'(task|todo|number|#)\s*(\d+)', text.lower())
        if match:
            params["todo_id"] = int(match.group(2))

        # Extract new title
        if "to" in text.lower():
            parts = text.lower().split("to", 1)
            if len(parts) > 1:
                new_title = parts[1].strip()
                new_title = re.sub(r'^(be|say)\s+', '', new_title)
                if new_title:
                    params["title"] = new_title

        # Extract new priority
        priority = self._extract_priority(text)
        if priority:
            params["priority"] = priority

        return params

    def _extract_priority(self, text: str) -> Optional[str]:
        """Extract priority from text"""
        text_lower = text.lower()

        if any(word in text_lower for word in self.PRIORITY_HIGH):
            return "high"
        elif any(word in text_lower for word in self.PRIORITY_LOW):
            return "low"
        elif "medium" in text_lower:
            return "medium"

        return None

    def _extract_tags(self, text: str) -> List[str]:
        """Extract tags from text"""
        # Look for hashtags
        hashtags = re.findall(r'#(\w+)', text)

        # Look for "tagged/categorized as X"
        match = re.search(r'tagged?\s+(?:as\s+)?(.+?)(?:\s+and|\s*$)', text.lower())
        if match:
            tags_text = match.group(1)
            tags = [t.strip() for t in tags_text.split(',')]
            hashtags.extend(tags)

        return list(set(hashtags))  # Remove duplicates

    def _extract_due_date(self, text: str) -> Optional[datetime]:
        """Extract due date from natural language"""
        text_lower = text.lower()

        # Today
        if "today" in text_lower:
            return datetime.now().replace(hour=23, minute=59, second=59)

        # Tomorrow
        if "tomorrow" in text_lower:
            return (datetime.now() + timedelta(days=1)).replace(hour=23, minute=59, second=59)

        # Next week
        if "next week" in text_lower:
            return (datetime.now() + timedelta(weeks=1)).replace(hour=23, minute=59, second=59)

        # In X days
        match = re.search(r'in (\d+) days?', text_lower)
        if match:
            days = int(match.group(1))
            return (datetime.now() + timedelta(days=days)).replace(hour=23, minute=59, second=59)

        # Specific date (YYYY-MM-DD)
        match = re.search(r'(\d{4}-\d{2}-\d{2})', text)
        if match:
            try:
                return datetime.fromisoformat(match.group(1))
            except ValueError:
                pass

        return None
```

### 4. Conversation State Management (Stateless Server)

**Primary Responsibility**: Manage conversation context in stateless manner using backend database.

**Conversation Manager** (`chatbot/src/conversation/manager.py`):

```python
"""
Conversation state management
"""

from typing import List, Dict, Any, Optional
from datetime import datetime
import httpx

class ConversationManager:
    """Manage conversation state with backend API"""

    def __init__(self, backend_url: str, auth_token: str):
        self.backend_url = backend_url
        self.auth_token = auth_token
        self.client = httpx.AsyncClient(
            base_url=backend_url,
            headers={"Authorization": f"Bearer {auth_token}"}
        )

    async def get_conversation(
        self,
        user_id: int,
        conversation_id: Optional[int] = None
    ) -> Dict[str, Any]:
        """
        Get conversation history from backend

        Args:
            user_id: User ID
            conversation_id: Optional conversation ID (creates new if None)

        Returns:
            Conversation object with messages
        """
        if conversation_id:
            # Get existing conversation
            response = await self.client.get(f"/api/conversations/{conversation_id}")
            return response.json()
        else:
            # Create new conversation
            response = await self.client.post(
                "/api/conversations",
                json={"user_id": user_id}
            )
            return response.json()

    async def add_message(
        self,
        conversation_id: int,
        role: str,
        content: str,
        metadata: Optional[Dict[str, Any]] = None
    ) -> Dict[str, Any]:
        """
        Add message to conversation

        Args:
            conversation_id: Conversation ID
            role: 'user' or 'assistant'
            content: Message text
            metadata: Optional metadata (tools used, etc.)

        Returns:
            Created message object
        """
        response = await self.client.post(
            "/api/messages",
            json={
                "conversation_id": conversation_id,
                "role": role,
                "content": content,
                "metadata": metadata or {}
            }
        )
        return response.json()

    async def get_messages(
        self,
        conversation_id: int,
        limit: int = 50
    ) -> List[Dict[str, Any]]:
        """Get recent messages from conversation"""
        response = await self.client.get(
            f"/api/conversations/{conversation_id}/messages",
            params={"limit": limit}
        )
        return response.json()["messages"]

    async def get_context_for_agent(
        self,
        conversation_id: int,
        max_messages: int = 10
    ) -> str:
        """
        Get formatted conversation context for Claude agent

        Returns conversation history as formatted string
        """
        messages = await self.get_messages(conversation_id, limit=max_messages)

        # Format as conversation
        context_lines = []
        for msg in messages:
            role = "User" if msg["role"] == "user" else "Assistant"
            context_lines.append(f"{role}: {msg['content']}")

        return "\n".join(context_lines)
```

**Stateless Design**:
- Chatbot server does not store conversation state in memory
- All conversation history stored in backend database (Conversation, Message tables)
- Each request fetches recent conversation context from backend
- Enables horizontal scaling of chatbot servers
- Survives server restarts without losing conversations

```python
"""
Language detection and translation
"""

from typing import Optional
import re

class LanguageDetector:
    """Detect input language and translate if needed"""

    # Urdu Unicode range
    URDU_PATTERN = re.compile(r'[\u0600-\u06FF]')

    def detect_language(self, text: str) -> str:
        """
        Detect language of input text

        Returns: 'en' (English), 'ur' (Urdu), or 'unknown'
        """
        # Check for Urdu characters
        if self.URDU_PATTERN.search(text):
            return 'ur'

        # Default to English
        return 'en'

    async def translate_to_english(self, text: str, source_lang: str) -> str:
        """
        Translate text to English for MCP tool processing

        Args:
            text: Input text
            source_lang: Source language code

        Returns:
            English translation
        """
        if source_lang == 'en':
            return text

        # Use translation API (e.g., Google Translate, DeepL)
        # For Urdu, could use specialized Urdu-English translation
        from googletrans import Translator

        translator = Translator()
        result = translator.translate(text, src=source_lang, dest='en')
        return result.text

    async def translate_from_english(self, text: str, target_lang: str) -> str:
        """
        Translate English response to target language

        Args:
            text: English text
            target_lang: Target language code

        Returns:
            Translated text
        """
        if target_lang == 'en':
            return text

        from googletrans import Translator

        translator = Translator()
        result = translator.translate(text, src='en', dest=target_lang)
        return result.text
```

**Multilingual Agent Wrapper** (`chatbot/src/agents/multilingual_agent.py`):

```python
"""
Multilingual wrapper for TodoAgent
"""

from .todo_agent import TodoAgent, AgentResponse
from ..nlp.language_detector import LanguageDetector

class MultilingualTodoAgent:
    """Multilingual wrapper for TodoAgent"""

    def __init__(self, user_id: int, conversation_id: Optional[int] = None):
        self.todo_agent = TodoAgent(user_id, conversation_id)
        self.language_detector = LanguageDetector()

    async def process(self, user_input: str) -> AgentResponse:
        """
        Process user input in any supported language

        Args:
            user_input: User's message in any language

        Returns:
            AgentResponse with text in user's language
        """
        # Detect language
        user_lang = self.language_detector.detect_language(user_input)

        # Translate to English if needed
        if user_lang != 'en':
            english_input = await self.language_detector.translate_to_english(
                user_input,
                source_lang=user_lang
            )
        else:
            english_input = user_input

        # Process with TodoAgent (in English)
        response = await self.todo_agent.process(english_input)

        # Translate response back to user's language
        if user_lang != 'en':
            translated_text = await self.language_detector.translate_from_english(
                response.text,
                target_lang=user_lang
            )
            response.text = translated_text

        return response
```
